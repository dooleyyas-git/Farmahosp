
-- 1. TABELAS BASE (sem FK)


CREATE TABLE PESSOAS (
    id INT NOT NULL,
    nome VARCHAR(255) NOT NULL,
    cpf VARCHAR(14) UNIQUE,
    CONSTRAINT PK_PESSOAS PRIMARY KEY (id)
);

CREATE TABLE LABORATORIOS (
    id INT NOT NULL,
    nome VARCHAR(255) NOT NULL,
    descricao TEXT,
    CONSTRAINT PK_LABORATORIOS PRIMARY KEY (id)
);

CREATE TABLE MEDICAMENTOS (
    id INT NOT NULL,
    nome VARCHAR(255) NOT NULL,
    descricao TEXT,
    dosagem VARCHAR(50),
    unidade VARCHAR(20),
    quantidade_caixa INT,
    tipo VARCHAR(50),
    quantidade_atual INT DEFAULT 0,
    alta_prioridade INT DEFAULT 0, -- Adicionado para o controle de estoque do Flask
    CONSTRAINT PK_MEDICAMENTOS PRIMARY KEY (id)
);

CREATE TABLE UTENSILIOS (
    id INT NOT NULL,
    nome VARCHAR(255) NOT NULL,
    descricao TEXT,
    quantidade_atual INT DEFAULT 0,
    CONSTRAINT PK_UTENSILIOS PRIMARY KEY (id)
);

-- 2. TABELAS DE HERANÇA (Relacionamento 1:1 com PESSOAS)


CREATE TABLE FUNCIONARIOS (
    pessoa_id INT NOT NULL,
    senha VARCHAR(255) NOT NULL,
    CONSTRAINT PK_FUNCIONARIOS PRIMARY KEY (pessoa_id),
    CONSTRAINT FK_FUNCIONARIOS_PESSOAS FOREIGN KEY (pessoa_id)
        REFERENCES PESSOAS(id) ON DELETE CASCADE
);

CREATE TABLE MEDICOS (
    pessoa_id INT NOT NULL,
    crm VARCHAR(50) UNIQUE NOT NULL,
    CONSTRAINT PK_MEDICOS PRIMARY KEY (pessoa_id),
    CONSTRAINT FK_MEDICOS_PESSOAS FOREIGN KEY (pessoa_id)
        REFERENCES PESSOAS(id) ON DELETE CASCADE
);

CREATE TABLE ENFERMEIROS (
    pessoa_id INT NOT NULL,
    coren VARCHAR(50) UNIQUE NOT NULL,
    CONSTRAINT PK_ENFERMEIROS PRIMARY KEY (pessoa_id),
    CONSTRAINT FK_ENFERMEIROS_PESSOAS FOREIGN KEY (pessoa_id)
        REFERENCES PESSOAS(id) ON DELETE CASCADE
);

CREATE TABLE PACIENTES (
    pessoa_id INT NOT NULL,
    CONSTRAINT PK_PACIENTES PRIMARY KEY (pessoa_id),
    CONSTRAINT FK_PACIENTES_PESSOAS FOREIGN KEY (pessoa_id)
        REFERENCES PESSOAS(id) ON DELETE CASCADE
);


-- 3. TABELAS DE ATENDIMENTO E LOGÍSTICA (Relacionamentos 1:N e N:N)


CREATE TABLE MEDICAMENTO_LABORATORIO (
    id_medicamento INT NOT NULL,
    id_laboratorio INT NOT NULL,
    CONSTRAINT PK_MEDICAMENTO_LABORATORIO PRIMARY KEY (id_medicamento, id_laboratorio),
    CONSTRAINT FK_MED_LAB_MEDICAMENTOS FOREIGN KEY (id_medicamento)
        REFERENCES MEDICAMENTOS(id) ON DELETE CASCADE,
    CONSTRAINT FK_MED_LAB_LABORATORIOS FOREIGN KEY (id_laboratorio)
        REFERENCES LABORATORIOS(id) ON DELETE CASCADE
);

CREATE TABLE RECEITAS (
    id INT NOT NULL,
    paciente_id INT NOT NULL,
    medico_id INT NOT NULL,
    data_emissao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT PK_RECEITAS PRIMARY KEY (id),
    CONSTRAINT FK_RECEITAS_PACIENTES FOREIGN KEY (paciente_id)
        REFERENCES PACIENTES(pessoa_id),
    CONSTRAINT FK_RECEITAS_MEDICOS FOREIGN KEY (medico_id)
        REFERENCES MEDICOS(pessoa_id)
);

CREATE TABLE ITENS_RECEITA (
    receita_id INT NOT NULL,
    medicamento_id INT NOT NULL,
    quantidade INT NOT NULL,
    CONSTRAINT PK_ITENS_RECEITA PRIMARY KEY (receita_id, medicamento_id),
    CONSTRAINT FK_ITENS_RECEITA_RECEITAS FOREIGN KEY (receita_id)
        REFERENCES RECEITAS(id) ON DELETE CASCADE,
    CONSTRAINT FK_ITENS_RECEITA_MEDICAMENTOS FOREIGN KEY (medicamento_id)
        REFERENCES MEDICAMENTOS(id)
);


-- 4. TABELAS DE MOVIMENTAÇÃO E AUDITORIA


CREATE TABLE MOVIMENTACOES_MEDICAMENTOS (
    id INT NOT NULL,
    id_medicamento INT NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    quantidade INT NOT NULL,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT PK_MOV_MEDICAMENTOS PRIMARY KEY (id),
    CONSTRAINT CK_MOV_MED_TIPO CHECK (tipo IN ('entrada', 'saida')),
    CONSTRAINT FK_MOV_MED_MEDICAMENTOS FOREIGN KEY (id_medicamento)
        REFERENCES MEDICAMENTOS(id)
);

CREATE TABLE MOVIMENTACOES_UTENSILIOS (
    id INT NOT NULL,
    id_utensilio INT NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    quantidade INT NOT NULL,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT PK_MOV_UTENSILIOS PRIMARY KEY (id),
    CONSTRAINT CK_MOV_UTEN_TIPO CHECK (tipo IN ('entrada', 'saida')),
    CONSTRAINT FK_MOV_UTEN_UTENSILIOS FOREIGN KEY (id_utensilio)
        REFERENCES UTENSILIOS(id)
);

CREATE TABLE HISTORICO (
    id INT NOT NULL,
    id_medicamento INT NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    quantidade INT NOT NULL,
    data_hora VARCHAR(20) NOT NULL, -- Mantido texto para bater com o formato do Flask
    CONSTRAINT PK_HISTORICO PRIMARY KEY (id),
    CONSTRAINT FK_HISTORICO_MEDICAMENTOS FOREIGN KEY (id_medicamento)
        REFERENCES MEDICAMENTOS(id)
);
