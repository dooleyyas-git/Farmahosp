
CREATE TABLE FUNCIONARIOS (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    cpf VARCHAR(20) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL
);

CREATE TABLE MEDICAMENTOS (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    descricao VARCHAR(500),
    dosagem VARCHAR(50),
    unidade VARCHAR(50),
    quantidade_caixa INTEGER,
    tipo VARCHAR(50),
    quantidade_atual INTEGER DEFAULT 0
);

CREATE TABLE UTENSILIOS (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    descricao VARCHAR(500),
    quantidade_atual INTEGER DEFAULT 0
);

CREATE TABLE LABORATORIOS (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    descricao VARCHAR(500)
);

CREATE TABLE MEDICAMENTO_LABORATORIO (
    id_medicamento INTEGER NOT NULL,
    id_laboratorio INTEGER NOT NULL,
    PRIMARY KEY (id_medicamento, id_laboratorio),
    FOREIGN KEY (id_medicamento) REFERENCES MEDICAMENTOS(id),
    FOREIGN KEY (id_laboratorio) REFERENCES LABORATORIOS(id)
);

CREATE TABLE MEDICOS (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    crm VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE ENFERMEIROS (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    coren VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE PACIENTES (
    id INTEGER PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    cpf VARCHAR(20) UNIQUE
);

CREATE TABLE MOVIMENTACOES_UTENSILIOS (
    id INTEGER PRIMARY KEY,
    id_utensilio INTEGER NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CHECK (tipo IN ('entrada', 'saida')),
    FOREIGN KEY (id_utensilio) REFERENCES UTENSILIOS(id)
);

CREATE TABLE MOVIMENTACOES_MEDICAMENTOS (
    id INTEGER PRIMARY KEY,
    id_medicamento INTEGER NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CHECK (tipo IN ('entrada', 'saida')),
    FOREIGN KEY (id_medicamento) REFERENCES MEDICAMENTOS(id)
);
